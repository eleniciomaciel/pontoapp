HTML structure

Sangoma template is a fixed layout container with full 12col Bootstrap grid.

The best way how to get familiar with each element is to take a look at code itself. Every party is well documented and in combination with origianl Bootstrap documentation you should't get lost.

Standard layout

<div id="wrapper">

	<!-- Main page header -->
	<header id="header" class="container">

		<h1>

			<!-- Main page logo -->
			<a href="#">Logo</a>

			<p><!-- page headline --></p>

		</h1>


		<!-- User profile -->
		<div class="user-profile">

			...

		</div>

		<!-- Main navigation -->
		<nav class="main-navigation-collapse collapse navbar-collapse" role="navigation">

			...

		</nav>

	</header>
	<!-- /Main page header -->

	<!-- Main page container -->
	<section class="container" role="main">

		<!-- Page header -->
		<article class="page-header">
			...
		</article>

		<!-- Grid row -->
		<div class="row">

			<!-- Data block -->
			<article class="col-sm-X">
				<div class="data-block">
					<header>
						...
					</header>
					<section>
						...
					</section>
					<footer>
						...
					</footer>
				</div>
			</article>

		</div>

	</section>
	<!-- /Main page container -->

	<!-- Sticky footer push -->
	<div id="push"></div>

</div>
<!-- Full height wrapper -->

<!-- Main page footer -->
<footer id="footer" class="container">
	...
</footer>
